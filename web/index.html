Quick start

# Random roast
curl https://roastapi.vercel.app/api/roast

# Roast a developer at brutal level
curl "https://roastapi.vercel.app/api/roast/developer?level=brutal&style=gordon_ramsay"

# Roast your code
curl -X POST https://roastapi.vercel.app/api/roast/code \
  -H "Content-Type: application/json" \
  -d '{"code": "for i in range(len(arr)):"}'
import requests

# Roast a developer
r = requests.get("https://roastapi.vercel.app/api/roast/developer",
               params={"level": "savage", "style": "gordon_ramsay"})
data = r.json()
print(data["roast"])           # the roast
print(data["damage_rating"])    # e.g. 8.4

# Personalised roast
r = requests.post("https://roastapi.vercel.app/api/roast/me",
                  json={"name": "Chad", "job": "10x Engineer"})
print(r.json()["roast"])
// Roast by category
const res = await fetch(
  `https://roastapi.vercel.app/api/roast/founder?level=brutal`
);
const { roast, damage_rating } = await res.json();
console.log(roast, `(${damage_rating}/10)`);

// Roast some code
const r = await fetch("https://roastapi.vercel.app/api/roast/code", {
  method: "POST",
  headers: { "Content-Type": "application/json" },
  body: JSON.stringify({ code: "var x = new Array()" }),
});
print((await r.json()).roast);
